Checkout Android AOSP
$ source build/envsetup.sh
$ lunch full-eng
$ export SOONG_ALLOW_MISSING_DEPENDENCIES=true
$ time m -j16 build-art-host

Delete everything in the art folder via 'rm -fr'
except for the BUILD file, copy the output from ART.

cd out/host/linux-x86
art=<YOUR_PREBUILT_IN_STUDIO_MAIN_REPO>/tools/linux-x86_64/art
mkdir $art/bin
cp bin/art $art/bin
cp bin/dalvikvm64 $art/bin
cp bin/dex2oatd $art/bin
cp bin/dex2oat $art/bin
cp -r lib64 ~/repos/studio-main/prebuilts/tools/linux-x86_64/art/.
cp -r framework ~/repos/studio-main/prebuilts/tools/linux-x86_64/art/.
cp -r com.android.i18n ~/repos/studio-main/prebuilts/tools/linux-x86_64/art/.
mkdir $art/usr
cp -r usr/share ~/repos/studio-main/prebuilts/tools/linux-x86_64/art/usr/.
